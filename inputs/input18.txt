package main;

import "fmt";

func res(input int) int {
    var temp, b, final int;
    temp = input + 10;
    b = 5 * 2 + 3 - 2;
    final = temp - b;
    return final
}

func main() {
	var result int;
	result = res(50);
    fmt.Println(result)
}

// Expected output: 49
