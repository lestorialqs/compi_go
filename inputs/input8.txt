package main

import "fmt"

type Person struct {
	firstName string;
	lastName  string;
	age       int
}

func main() {
	var p1 Person;
	p1.firstName = "Homero";
	p1.lastName = "Simpson";
	p1.age = 38;

	var p2 Person;
	p2.firstName = "Ned";
	p2.lastName = "Flanders";
	p2.age = 36;

	fmt.Println(p1.firstName);
	fmt.Println(p2.lastName);
	fmt.Println(p2.age)
}

// Expected output: Homero, Flanders, 36
