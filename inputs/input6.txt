package main;

import "fmt";

type Student struct {
	name string;
	age int
}

func main() {
	var s1 Student;
	var s2 Student;

	s1.name = "Pedro";
	s1.age = 12;

	s2.name = "Alfredo";
	s2.age = 30;

	fmt.Println((s1.age + s2.age)*2);
	fmt.Println(s1.name);
	fmt.Println(s2.name)
}

// Expected output: 84, Pedro, Alfredo
